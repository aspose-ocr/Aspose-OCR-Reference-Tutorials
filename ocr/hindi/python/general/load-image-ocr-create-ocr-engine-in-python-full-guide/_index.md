```python
# load_image_ocr_demo.py
import os
import ocr

def init_engine():
    try:
        engine = ocr.OcrEngine()
        print("тЬЕ OCR engine created.")
        return engine
    except ocr.OcrException as e:
        print(f"тЭМ Could not create OCR engine (code {e.code}): {e.message}")
        raise

def load_image_with_exception(engine, path):
    try:
        engine.load_image(path)
        print(f"тЬЕ Loaded image via exception method: {path}")
    except ocr.OcrException as ex:
        print(f"тЭМ Exception while loading '{path}': {ex.message}")

def load_image_with_error_api(engine, path):
    engine.load_image(path)
    err = engine.get_last_error()
    if err:
        print(f"тЭМ Error API reported for '{path}': {err.message}")
    else:
        print(f"тЬЕ Loaded image via error API: {path}")

def extract_text(engine):
    try:
        result = engine.recognize()
        txt = result.text
        if txt:
            print("ЁЯУЭ OCR Result:")
            print(txt)
        else:
            print("тЪая╕П No recognizable text.")
        return txt
    except ocr.OcrException as e:
        print(f"тЭМ Recognition error: {e.message}")
        return ""

def main():
    # 1я╕ПтГг Create the OCR engine
    engine = init_engine()

    # Paths тАУ adjust to your environment
    existing_img = os.path.join("samples", "document.png")
    missing_img = os.path.join("samples", "nonexistent.png")

    # 2я╕ПтГг Load a valid image using exception handling
    load_image_with_exception(engine, existing_img)
    extract_text(engine)

    # 3я╕ПтГг Attempt to load a missing image using the error API
    load_image_with_error_api(engine, missing_img)

if __name__ == "__main__":
    main()
```
